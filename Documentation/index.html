EduBright API Documentation

Register

POST /register

Request Body

{
    "name": "kuraimos",
    "email": "kuraimoss@edubright.com",
    "password": "password123"
}

Response - Success

{
    "status": "success",
    "message": "User registered successfully!",
    "registerResult": {
        "userId": "user-12345",
        "name": "kuraimos",
        "email": "kuraimoss@edubright.com"
    }
}

Response - Error

{
    "status": "error",
    "error": "The email address is already registered.",
    "details": "Duplicate entry 'kuraimoss@edubright.com' for key 'email'",
    "registerResult": {
        "userId": null,
        "name": "kuraimos",
        "email": "kuraimoss@edubright.com"
    }
}
Login

POST /login

Request Body

{
    "email": "kuraimoss@edubright.com",
    "password": "password123"
}

Response - Success

{
    "status": "success",
    "message": "Login successful!",
    "loginResult": {
        "token": "jwt_token_here",
        "userId": "user-12345",
        "name": "kuraimos",
        "email": "kuraimoss@edubright.com"
    }
}

Response - Error

{
    "status": "error",
    "error": "User not found. Please check the email address and try again.",
    "details": "No user found with the provided email."
}

{
    "status": "error",
    "error": "Invalid credentials. Please check your password and try again.",
    "details": "The provided password is incorrect."
}
Sentiment Prediction

POST /predict

Request Body

{
    "text": "I love this product! It's amazing."
}

Response - Success

{
    "success": true,
    "prediction": "Awesome",
    "timestamp": "2024-12-03T10:00:00.000Z",
    "message": "Prediksi berhasil diproses!"
}

Response - Error

{
    "success": false,
    "message": "An error occurred while processing the prediction",
    "timestamp": "2024-12-03T10:05:00.000Z"
}
