EduBright API Documentation

Register

POST /register

Request Body

{
    "name": "string",
    "email": "string", // must be unique
    "password": "string" // at least 6 characters
}

Response - Success

{
    "status": "success",
    "message": "User registered successfully!",
    "registerResult": {
        "userId": "user-xxxxxx",
        "name": "John Doe",
        "email": "john.doe@example.com"
    }
}

Response - Error

{
    "status": "error",
    "error": "The email address is already registered.",
    "details": "Duplicate entry 'john.doe@example.com' for key 'email'",
    "registerResult": {
        "userId": null,
        "name": "John Doe",
        "email": "john.doe@example.com"
    }
}
Login

POST /login

Request Body

{
    "email": "string",
    "password": "string"
}

Response - Success

{
    "status": "success",
    "message": "Login successful!",
    "loginResult": {
        "token": "jwt_token_here",
        "userId": "user-xxxxxx",
        "name": "John Doe",
        "email": "john.doe@example.com"
    }
}

Response - Error

{
    "status": "error",
    "error": "User not found. Please check the email address and try again.",
    "details": "No user found with the provided email."
}

{
    "status": "error",
    "error": "Invalid credentials. Please check your password and try again.",
    "details": "The provided password is incorrect."
}
